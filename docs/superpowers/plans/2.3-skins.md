### skin store

```js
// client/src/stores/skin.js
import { defineStore } from 'pinia'
import { SKINS } from '../lib/skins.js'

export const useSkinStore = defineStore('skin', {
  state: () => ({
    activeSkinId: localStorage.getItem('activeSkin') || 'default',
    entitlements: []  // 4 迭代时由 /api/me/entitlements 填充；本迭代保持空数组
  }),
  getters: {
    activeSkin: (state) => SKINS[state.activeSkinId] || SKINS.default,
    availableSkins: (state) => {
      return Object.values(SKINS).filter(skin =>
        skin.free || state.entitlements.includes(skin.entitlementId)
      )
    },
    allSkins: () => Object.values(SKINS)  // 列表页全显示（锁定态显示 🔒）
  },
  actions: {
    setActive(skinId) {
      const skin = SKINS[skinId]
      if (!skin) return
      if (!skin.free && !this.entitlements.includes(skin.entitlementId)) return
      this.activeSkinId = skinId
      localStorage.setItem('activeSkin', skinId)
    }
  }
})
```
